war_stuff.1 = {
	type = country_event
	placement = scope:war_stuff_state

	title = war_stuff.1.t
	desc = war_stuff.1.d
	flavor = war_stuff.1.f

	event_image = {
		video = "unspecific_military_parade"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/waving_flag.dds"

	cooldown = { days = 1 }

	duration = 3

	trigger = {
		any_scope_war = {
			any_scope_front = {
				any_military_formation = {
					owner = ROOT
				}
			}
		}
		any_enemy_in_war = {
			country_rank >= root.country_rank
			capital = {
				NOT = { has_variable = occupied }
				controller = {
					this = root
				}
			}
		}
	}

	immediate = {
		ROOT = { save_scope_as = war_stuff_home }
		
		random_enemy_in_war = {
			limit = {
				capital = {
					controller = {
						this = root
					}
				}
				country_rank >= root.country_rank
			}
			save_scope_as = war_stuff_enemy
			capital = {
				save_scope_as = war_stuff_state
				set_variable = occupied
			}
			random_primary_culture = {
				save_scope_as = war_stuff_culture
			}
		}
		random_scope_war = {
			limit = {
				any_scope_front = {
					any_military_formation = {
						owner = ROOT
					}
				}
			}
			save_scope_as = war_stuff_war
			random_scope_front = {
				limit = {
					any_military_formation = {
						owner = ROOT
					}
				}
				random_scope_general = {
					limit = {
						owner = ROOT
						commander_military_formation = {
							owner = ROOT
						}
					}
					save_scope_as = war_stuff_general
				}
			}
		}
	}

	option = { # leave them
		name = war_stuff.1.a
		default_option = yes

		scope:war_stuff_general = {
			add_trait = honorable
		}
		change_relations = {
			country = scope:war_stuff_enemy
			value = 5
		}
		scope:war_stuff_enemy = {
			trigger_event = { id = war_stuff.2 }
		}
		every_political_lobby = {
			limit = {
				target = scope:war_stuff_enemy
				is_political_lobby_type = lobby_pro_country
			}
			change_appeasement = {
				amount = 5
				factor = appeasement_special_events_positive
			}
		}
		every_political_lobby = {
			limit = {
				target = scope:war_stuff_enemy
				is_political_lobby_type = lobby_anti_country
			}
			change_appeasement = {
				amount = -5
				factor = appeasement_special_events_negative
			}
		}
		ai_chance = {
			base = 60
		}
	}

	option = { # kill
		name = war_stuff.1.b

		scope:war_stuff_state = {
			state_region = {
				add_devastation = 40
			}
			kill_population_percent_in_state = {
				percent = 0.1
				culture = scope:war_stuff_culture
			}
		}
		scope:war_stuff_general = {
			add_trait = cruel
		}
		change_infamy = 15
		set_variable = waddle_recent_atrocity
		change_relations = {
			country = scope:war_stuff_enemy
			value = -30
		}
		scope:war_stuff_enemy = {
			trigger_event = { id = war_stuff.3 }
		}
		every_political_lobby = {
			limit = {
				target = scope:war_stuff_enemy
				is_political_lobby_type = lobby_pro_country
			}
			change_appeasement = {
				amount = -5
				factor = appeasement_special_events_negative
			}
		}
		every_political_lobby = {
			limit = {
				target = scope:war_stuff_enemy
				is_political_lobby_type = lobby_anti_country
			}
			change_appeasement = {
				amount = 5
				factor = appeasement_special_events_positive
			}
		}
		ai_chance = {
			base = 40
		}
	}
}

war_stuff.13 = {
	type = country_event
	title = war_stuff.13.t
	desc = war_stuff.13.d
	flavor = war_stuff.13.f

	cooldown = { days = long_modifier_time }
	duration = 3
	
	event_image = {
		video = "europenorthamerica_before_the_battle"
	}

	trigger = {
		is_at_war = yes
		any_scope_war = {
			diplomatic_play = {
				var:waddle_war_length > 2 # At least 2 years of war
				var:waddle_war_balance_change > -0.05
				var:waddle_war_balance_change < 0.05
			}
			any_scope_front = {
				any_military_formation = {
					owner = ROOT
				}
			}
		}
		any_military_formation = {
			count = all
			is_fully_mobilized = yes
		}
	}

	immediate = {
		random_scope_war = {
			save_scope_as = war_stuff_war
			limit = {
				diplomatic_play = {
					var:waddle_war_length > 2 # At least 2 years of war
					var:waddle_war_balance_change > -0.05
					var:waddle_war_balance_change < 0.05
				}
				any_scope_front = {
					any_military_formation = {
						owner = ROOT
					}
				}
			}
			random_scope_front = {
				limit = {
					any_military_formation = {
						owner = ROOT
					}
				}
				random_scope_state = {
					limit = {
						NOT = {
							owner = {
								this = ROOT
							}
						}
					}
					owner = {
						save_scope_as = war_stuff_enemy
					}
				}
				save_scope_as = war_stuff_front
				random_scope_general = {
					limit = {
						owner = ROOT
						commander_military_formation = {
							owner = ROOT
						}
					}
					save_scope_as = war_stuff_general_offensive
				}
			}
		}
	}

	option = {
		name = war_stuff.13.a # We must break the deadlock!
		
		# High risk, high reward
		add_modifier = {
			name = waddle_modifier_desperate_offensive
			months = 6
			is_decaying = yes
		}
		set_variable = waddle_recent_desperate_offensive
		scope:war_stuff_war = {
			diplomatic_play = {
				set_variable = waddle_recent_desperate_offensive
			}
		}
		
		# Significant manpower cost for the "Big Push"
		add_radicals = {
			value = 0.02
			pop_type = soldiers
		}
	}

	option = {
		name = war_stuff.13.b # Dig in and wait
		default_option = yes
		
		add_modifier = {
			name = waddle_modifier_trench_attrition
			months = 12
			is_decaying = yes
		}
	}
}

# Liberation Offensive (Reclaiming Owned States)
war_stuff.14 = {
    type = country_event
    title = war_stuff.14.t
    desc = war_stuff.14.d
    flavor = war_stuff.14.f

    cooldown = { days = long_modifier_time }
    duration = 3
    
    event_image = {
        video = "europenorthamerica_before_the_battle"
    }

    trigger = {
        is_at_war = yes
        any_scope_war = {
            # Check for stalemate
            diplomatic_play = {
                var:waddle_war_length > 1
                var:waddle_war_balance_change > -0.05
                var:waddle_war_balance_change < 0.05
            }
            # The Front must contain an incorporated state owned by ROOT but NOT controlled by ROOT
			any_scope_front = {
				any_military_formation = { owner = ROOT }
				any_scope_state = {
					owner = ROOT
					is_incorporated = yes
					NOT = {
						controller = {
							this = root
						}
					}
				}
			}
        }
        any_military_formation = {
			count = all
			is_fully_mobilized = yes
		}
    }

    immediate = {
        random_scope_war = {
            save_scope_as = war_stuff_war
            limit = {
                any_scope_front = {
                    any_military_formation = { owner = ROOT }
                    any_scope_state = {
                        is_incorporated = yes
						owner = ROOT
                        NOT = {
							controller = {
								this = root
							}
						}
                    }
                }
            }
            random_scope_front = {
                limit = {
                    any_military_formation = { owner = ROOT }
                    any_scope_state = {
                        is_incorporated = yes
                        owner = ROOT
                        NOT = {
							controller = {
								this = root
							}
						}
                    }
                }
                save_scope_as = war_stuff_front
				random_scope_state = {
                    limit = {
						is_incorporated = yes
                        owner = ROOT
                        NOT = {
							controller = {
								this = root
							}
						}
					}
					save_scope_as = war_stuff_liberation_state
					controller = {
						save_scope_as = war_stuff_enemy
					}
                }

                random_scope_general = {
                    limit = {
                        owner = ROOT
                        commander_military_formation = { owner = ROOT }
                    }
                    save_scope_as = war_stuff_general_liberation
                }
            }
        }
    }

    option = {
        name = war_stuff.14.a # Liberate our people!
        
        add_modifier = {
            name = waddle_modifier_liberation_drive
            months = 6
            is_decaying = yes
        }
        set_variable = waddle_recent_liberation_offensive
        scope:war_stuff_war = {
            diplomatic_play = {
                set_variable = waddle_recent_liberation_offensive
            }
        }
        
    }

    option = {
        name = war_stuff.14.b # It is too risky.
        default_option = yes
        
        add_modifier = {
            name = waddle_modifier_trench_attrition
            months = 12
            is_decaying = yes
        }
        every_scope_state = {
			limit = {
				is_incorporated = yes
				owner = ROOT
                NOT = { controller = ROOT }
			}
			add_radicals_in_state = {
				value = 0.05
			}
		}
    }
}
