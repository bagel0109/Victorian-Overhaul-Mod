REPLACE_OR_CREATE:demilitarise = {
	icon = "gfx/interface/icons/law_icons/unused/revolutionary_conscription.dds"
	kind = custom
	execution_priority = 2001
	contestion_type = control_target_country_capital
	target_type = country
	possible = {
		OR = {
			has_game_rule = waddle_demilitarise_allowed
			AND = {
				is_player = yes
				has_game_rule = waddle_demilitarise_allowed_player
			}
		}
		root = {
			OR = {
				country_rank >= rank_value:major_power
			}
			
		}
		scope:target_country = {
			OR = {
				country_rank >= rank_value:major_power
			}
		}
	}
	maneuvers = {
		value = 0

		scope:target_country = {
			add = {
				value = country_rank
				multiply = 1.7
			}
		}
		if = {
			limit = {
				any_rival_country = {
					this = scope:target_country
				}
				
			}
			multiply = 0.5
		}
		min = 10
		max = 80
	}

	infamy = {
		value = 0

		scope:target_country = {
			add = {
				value = army_size_including_conscripts
				# divide = 20
			}
		}
		divide = {
			every_country = {
				limit = {
					country_rank >= rank_value:major_power
				}
				add = army_size_including_conscripts
			}
		}
		multiply = 200
		if = {
			limit = {
				any_rival_country = {
					this = scope:target_country
				}
				
			}
			multiply = 0.5
		}			
		
		min = 5
		max = 85.715
	}
	on_enforced = {
		if = {
			limit = {
				always = yes
				#i forgrgrrg
			}
			scope:target_country = {
				add_modifier = {
					name = demilitarised_modifier
					months = 60
				}
				save_scope_value_as = {
					name = mil_size
					value = {
						add = army_size #_including_conscripts
						multiply = 0.34
					}
				}
				while = {
					limit = {
						scope:target_country.army_size > scope:mil_size
					}
					scope:target_country = {
						random_scope_state = {
							limit = {
								has_building = building_barrack
								is_capital = no
							}
							remove_building = building_barrack
						}
					}
				}
				change_infamy = {
					value = scope:mil_size
					multiply = -0.3
				}
				save_scope_value_as = {
					name = mil_size
					value = {
						add = navy_size
						multiply = 0.34
					}
				}
				while = {
					limit = {
						scope:target_country.navy_size > scope:mil_size
					}
					scope:target_country = {
						random_scope_state = {
							limit = {
								has_building = building_naval_base
								is_capital = no
							}
							remove_building = building_naval_base
						}
					}
				}
			}
		}
	}
}
