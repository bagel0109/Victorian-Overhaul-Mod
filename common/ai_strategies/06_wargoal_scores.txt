					# USA/CSA are included as compact civil-war settlement exceptions.
					OR = {
						scope:target_country.country_rank >= rank_value:great_power

						AND = {
							exists = c:USA
							scope:target_country = { c:USA ?= this }
						}
						AND = {
							exists = c:CSA
							scope:target_country = { c:CSA ?= this }
						}
					}

					can_reach_target_country = yes

					# Must have a serious strategic motive. Pariah-level infamy is itself enough.
					OR = {
						scope:target_country.infamy >= infamy_threshold:pariah

						has_diplomatic_pact = {
							who = scope:target_country
							type = rivalry
						}
						has_secret_goal = {
							who = scope:target_country
							secret_goal = conquer
						}
						has_secret_goal = {
							who = scope:target_country
							secret_goal = dominate
						}
						scope:target_country.infamy >= infamy_threshold:notorious
					}

					# Must either have real dismantle payload, be pariah-level infamous,
					# or be a compact national-collapse exception.
					OR = {
						scope:target_country.infamy >= infamy_threshold:pariah

						scope:target_country = {
							num_subjects >= 2
						}

						scope:target_country = {
							any_scope_state = {
								count >= 5
								NOT = {
									is_homeland_of_country_cultures = scope:target_country
								}
							}
						}

						AND = {
							scope:target_country = {
								num_subjects >= 1
							}
							scope:target_country = {
								any_scope_state = {
									count >= 3
									NOT = {
										is_homeland_of_country_cultures = scope:target_country
									}
								}
							}
						}

						AND = {
							exists = c:GER
							scope:target_country = { c:GER ?= this }
						}
						AND = {
							exists = c:NGF
							scope:target_country = { c:NGF ?= this }
						}
						AND = {
							exists = c:USA
							scope:target_country = { c:USA ?= this }
						}
						AND = {
							exists = c:CSA
							scope:target_country = { c:CSA ?= this }
						}
					}
				}

				# Minimal baseline. The score comes from threat, infamy, and payload.
				add = 1

				# Pariah-level infamy: effectively guaranteed if valid.
				if = {
					limit = {
						scope:target_country.infamy >= infamy_threshold:pariah
					}
					add = 1000
				}

				# Better Cut Down to Size logic: infamy scales directly and is uncapped.
				add = {
					value = scope:target_country.infamy
					multiply = 5
				}

				# Strategic motive.
				if = {
					limit = {
						has_diplomatic_pact = {
							who = scope:target_country
							type = rivalry
						}
					}
					add = 20
				}

				if = {
					limit = {
						has_secret_goal = {
							who = scope:target_country
							secret_goal = conquer
						}
					}
					add = 10
				}

				if = {
					limit = {
						has_secret_goal = {
							who = scope:target_country
							secret_goal = dominate
						}
					}
					add = 10
				}

				# Subject payload. This naturally drops after a target has already been dismantled.
				add = {
					value = scope:target_country.num_subjects
					multiply = 8
				}

				# Territorial empire payload: states outside the target's own primary-culture homelands.
				add = {
					value = 0

					scope:target_country = {
						every_scope_state = {
							limit = {
								NOT = {
									is_homeland_of_country_cultures = scope:target_country
								}
							}
							add = 2
						}
					}
				}

				# Economic payload: share of target GDP in non-primary-culture homeland states.
				# This keeps already-dismantled / low-payload countries lower unless they are pariahs.
				add = {
					value = 0

					scope:target_country = {
						every_scope_state = {
							limit = {
								NOT = {
									is_homeland_of_country_cultures = scope:target_country
								}
							}
							add = this.gdp
						}
					}

					if = {
						limit = {
							scope:target_country.gdp > 0
						}
						divide = scope:target_country.gdp
					}

					multiply = 75
				}

				# Compact national-collapse payloads.
				# These prevent GER/NGF and USA/CSA from being undervalued merely because
				# most of their states are their own primary-culture homelands.
				if = {
					limit = {
						OR = {
							AND = {
								exists = c:GER
								scope:target_country = { c:GER ?= this }
							}
							AND = {
								exists = c:NGF
								scope:target_country = { c:NGF ?= this }
							}
						}
					}
					add = 75
				}

				if = {
					limit = {
						OR = {
							AND = {
								exists = c:USA
								scope:target_country = { c:USA ?= this }
							}
							AND = {
								exists = c:CSA
								scope:target_country = { c:CSA ?= this }
							}
						}
					}
					add = 75
				}

				# World-order importance. Uncapped, but gentle.
				add = {
					value = scope:target_country.prestige
					divide = 100
				}

				# Live crisis against the target makes maximal demands more plausible.
				if = {
					limit = {
						is_diplomatic_play_enemy_of = scope:target_country
					}
					multiply = 1.15
				}

				# Do not pile on while already fighting other great powers.
				# Pariah-level targets still remain effectively guaranteed.
				every_enemy_in_war = {
					limit = {
						country_rank >= rank_value:great_power
					}
					add = -8
				}

				# Relations and ideology matter, but mildly.
				multiply = {
					value = "scope:target_country.relations:root"
					multiply = -0.003
					add = 1
				}

				multiply = {
					value = "ai_ideological_opinion(scope:target_country)"
					multiply = -0.001
					add = 1
				}
			}

			min = 0
		}
		custom = {
			value = 0

			add = 10
			# Relations with target
			multiply = {
				value = "scope:target_country.relations:root"
				multiply = -0.01
				add = 1
				min = 0.1
				max = 2.0
			}

			# Ideological opinion of target
			multiply = {
				value = "ai_ideological_opinion(scope:target_country)"
				multiply = -0.05
				add = 1
				min = 0.1
				max = 2.0
			}

			min = 0
		}
		demilitarise = {
			value = 0

			if = {
				limit = {
					scope:target_country.infamy >= infamy_threshold:notorious
				}
				add = 10

				if = {
					limit = {
						scope:target_country.infamy >= infamy_threshold:pariah
					}
					add = 10
				}
			}
		}
	}
