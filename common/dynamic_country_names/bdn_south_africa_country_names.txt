###southafrica####
REPLACE_OR_CREATE:SAF = {

	dynamic_country_name = {
		name = dyn_c_saf_dominion
		adjective = SAF_ADJ
		is_main_tag_only = yes
		priority = 10000
		trigger = {
    		exists = c:GBR
   		    scope:actor ?= {
        		is_subject_of = c:GBR
        		is_subject_type = subject_type_dominion
    		}
		}
	}	

	dynamic_country_name = {
		name = dyn_c_saf_colony
		adjective = SAF_ADJ
		is_main_tag_only = yes
		priority = 10000
		trigger = {
    		exists = c:GBR
   		    scope:actor ?= {
        		is_subject_of = c:GBR
        		NOR = { 
					is_subject_type = subject_type_dominion 
					is_subject_type = subject_type_protectorate
				}
    		}
		}
	}
	
	dynamic_country_name = {
		name = SAF
		adjective = SAF_ADJ
		use_overlord_prefix = yes
		is_main_tag_only = yes
		priority = 100

		trigger = {
			coa_def_lesser_subject_trigger = yes
		}
	}
}
