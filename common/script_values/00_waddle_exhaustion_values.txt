# root = country

REPLACE_OR_CREATE:war_exhaustion_from_enemy_contested_wargoals = {
	# add = define:NWar|WAR_EXHAUSTION_CONTESTED_ENEMY_WARGOALS
	add = 8
	multiply = "enemy_contested_wargoals(scope:war)"
}

REPLACE_OR_CREATE:military_size = {
	add = army_size_including_conscripts
	add = navy_size
}

REPLACE_OR_CREATE:war_exhaustion_from_casualties = {
	if = {
		limit = {
			"scope:war.num_country_casualties(root)" > 0
			"scope:war.num_casualties" > 0
		}

		add = "scope:war.num_country_casualties(root)"
		multiply = "scope:war.num_country_casualties(root)"
		divide = "scope:war.num_casualties"

		if = {
			limit = { military_size > 0 }
			divide = {
				add = total_population_share
				min = 0.02
				multiply = 50
				multiply = military_size
			}
		}

		divide = 1000
		multiply = define:NWar|WAR_EXHAUSTION_KIA_FACTOR
		multiply = {
			add = define:NWar|WAR_EXHAUSTION_KIA_BASE_MULTIPLIER
			add = {
				add = 1
				#"size_weighted_lost_battles_fraction(scope:war)"
				multiply = define:NWar|WAR_EXHAUSTION_KIA_BATTLES_LOST_MULTIPLIER
			}
		}
	}
}

REPLACE_OR_CREATE:war_exhaustion_from_lobby_clout = {
	add = "lobby_war_opposition(scope:war)"
	multiply = define:NWar|WAR_EXHAUSTION_ANTI_WAR_LOBBY_CLOUT_FACTOR
}

REPLACE_OR_CREATE:war_support_from_lobby_clout = {
	add = "lobby_war_support(scope:war)"
	multiply = define:NWar|WAR_EXHAUSTION_PRO_WAR_LOBBY_CLOUT_FACTOR
}

REPLACE_OR_CREATE:war_exhaustion_from_country_turmoil = {
	add = country_turmoil
	multiply = define:NWar|WAR_EXHAUSTION_TURMOIL_FACTOR
}

REPLACE_OR_CREATE:war_support_from_fervor = {
	if = {
		limit = {
			scope:war = {
				OR = {
					is_target_of_wargoal_in_war = root
					# is_holder_of_wargoal_in_war = root
				}
			}
		}
		subtract = country_fervor_primary_culture
		multiply = define:NPops|CULTURAL_FERVOR_WAR_SUPPORT_EFFECT
	}
}

			}

			}
		}

			limit = {

				multiply = scope:temp_enemy_occupation
				multiply = 7
				desc = "WAR_EXHAUSTION_FROM_ENEMY_OCCUPATION_HIGH"
			
			}

				scope:temp_enemy_occupation < 0.7
			}

				multiply = scope:temp_enemy_occupation

				multiply = scope:temp_enemy_occupation
				multiply = 7
				desc = "WAR_EXHAUSTION_FROM_ENEMY_OCCUPATION_EXTREME"

			add = 0.25
			add = {

			}

			add = 1

				subtract = 2

				}
			}
			desc = "WADDLE_LONG_WAR_CASUALTY_MULT"
		}

		}
	}

		limit = { scope:temp_war_exhaustion_from_acceptance_of_dead >= 0.01 }
		add = {
			add = scope:temp_war_exhaustion_from_acceptance_of_dead
			multiply = 0.3
