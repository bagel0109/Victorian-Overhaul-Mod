						target_is = root

			desc = "WADDLE_NO_ONE_CARES"

	}

	if = {
		limit = {
			any_enemy_in_diplo_play = {
				root = {
					play_participant_has_war_goal_of_type_against = {
						type = return_state
						target = prev
					}
				}
			}
		}
		add = {
			value = 20
			desc = "WADDLE_LAND_RECLAMATION"
		}
	}

	if = {
		limit = {
			has_law_or_variant = law_type:law_isolationism

			value = 10
			desc = "[GetLawType('law_serfdom').GetName]"
		}

					pro_war_ig_trigger = yes

			every_interest_group = {
				limit = {
					anti_war_ig_trigger = yes
				}
				add = {
					add = ig_clout
					multiply = 15
				}

					limit = { ig_approval > happy }

				limit = {

				}
				add = {
					add = political_movement_support

				# }

		}
		desc = "WADDLE_ANTI_WAR_MOVEMENTS"
	}

	add = {
		value = loyalist_fraction
		multiply = 100
		desc = "WADDLE_PRO_WAR_LOYALISTS"
	}

	subtract = {
		value = radical_fraction
		multiply = 100

REPLACE_OR_CREATE:waddle_war_progress = {
	value = 0
	every_war_participant = {
	  	limit = { OR = { is_in_war_together = prev.diplomatic_play.initiator } }
		add = {
			add = prestige
			multiply = {
				add = 1
				subtract = enemy_occupation
			}
		}
	}
	divide = {
		every_war_participant = {
			limit = { OR = { is_in_war_together = prev.diplomatic_play.initiator } }
			add = prestige
		}
	}
	save_temporary_value_as = guh
	divide = {
		every_war_participant = {
			limit = { OR = { is_in_war_together = prev.diplomatic_play.target } }
			add = {
				add = prestige
				multiply = {
					add = 1
					subtract = enemy_occupation
				}
			}
		}
		divide = {
			every_war_participant = {
				limit = { OR = { is_in_war_together = prev.diplomatic_play.target } }
				add = prestige
			}
		}
		add = scope:guh
	}
}

REPLACE_OR_CREATE:waddle_war_progress_other_way = {
	value = 1
	subtract = waddle_war_progress
